const App = () => {
  const divRef = useRef(null);
  useEffect(() => {
    console.log(divRef.current);
    console.log(1);
    return () => {
      console.log(2);
    };
  }, []);
  useLayoutEffect(() => {
    console.log(divRef.current);
    console.log(3);
  }, []);
  return <div ref={divRef}></div>;
};
