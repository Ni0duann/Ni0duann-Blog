n秒内不会重新计时

```javascript
function throttle(fn, wait) {
  let timer = null;
  return function () {
    if (!timer) {
      timer = setTimeout(() => {
        fn.apply(this, arguments);
        timer = null;
      }, wait);
    }
  };
}

function test() {
  console.log('test');
}

const throttleTest = throttle(test, 5000);

setInterval(() => {
  throttleTest();
}, 5);
```
