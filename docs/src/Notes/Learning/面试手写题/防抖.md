n秒内不会重新计时

```javascript
function debounce(fn, wait) {
  let timer = null;
  return function () {
    if (timer) {
      clearTimeout(timer);
    }
    timer = setTimeout(() => {
      fn.apply(this, arguments);
    }, wait);
  };
}

function test() {
  console.log('test');
}

const debounceTest = debounce(test, 1000);

setInterval(() => {
  debounceTest();
}, 500);
```
