/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */

/**
 * @param {ListNode} headA
 * @param {ListNode} headB
 * @return {ListNode}
 */
var getIntersectionNode = function (headA, headB) {
  let map = new Set();
  let temp = headA;
  while (temp) {
    map.add(temp);
    temp = temp.next;
  }
  temp = headB;
  while (temp) {
    if (map.has(temp)) return temp;
    temp = temp.next;
  }
};
