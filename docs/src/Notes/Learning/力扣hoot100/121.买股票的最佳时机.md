/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  let ans = 0;
  let min = prices[0];
  for (const p of prices) {
    ans = Math.max(p - min, ans);
    min = Math.min(p, min);
  }
  return ans;
};
