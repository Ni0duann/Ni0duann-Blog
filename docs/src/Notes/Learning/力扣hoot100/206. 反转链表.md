pre 不断移到最后，
curr不断反转指向

```javascript
const reverseList = function (head) {
  const pre = null;
  const curr = head;
  while (curr) {
    const next = curr.next;
    curr.next = pre;
    pre = curr;
    curr = next;
  }
  return pre;
};
```
