var rightSideView = function (root) {
  let ans = [];
  function dfs(node, depth) {
    if (node === null) return;
    if (depth === ans.length) {
      ans.push(node.val);
    }
    dfs(node.right, depth + 1);
    dfs(node.left, depth + 1);
  }
  dfs(root, 0);
  return ans;
};
